```
* 类型：POST 
* content-type:application/json

```
{

	recordEnd:"2018-01-05",//外呼结束时间 下限

	type:"success",//success 查询成功明细  fail 查询失败明细
	pagination:{	//分页参数
		 current: 1,
         pageSize: 10
	}
}
